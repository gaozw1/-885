```C
#include <stdio.h>

#define N 52

int main() {
    int a[N] = {0}, i, j, up = 0;
    for (i = 2; i <= N; i++)       //从第2张牌开始
        for (j = i; j <= N; j++)   //2的倍数
            if (j % i == 0)
                a[j - 1] = !a[j - 1]; //a[j-1]就是a[1]也就是第2张牌
    for (i = 0; i < N; i++) {
        if (a[i] == 0)
            up += 1;
    }
    printf("有%d张牌正面朝上", up);
    return 0;
}
```
